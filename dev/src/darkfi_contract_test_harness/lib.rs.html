121/// Wallet instance for a single [`Holder`]
122pub struct Wallet {
123    /// Main holder keypair
124    pub keypair: Keypair,
125    /// Keypair for arbitrary token minting
126    pub token_mint_authority: Keypair,
127    /// Keypair for arbitrary contract deployment
128    pub contract_deploy_authority: Keypair,
129    /// Holder's [`Validator`] instance
130    pub validator: ValidatorPtr,
131    /// Holder's instance of the Merkle tree for the `Money` contract
132    pub money_merkle_tree: MerkleTree,
133    /// Holder's instance of the SMT tree for the `Money` contract
134    pub money_null_smt: SmtMemoryFp,
135    /// Holder's instance of the SMT tree for the `Money` contract (snapshotted for DAO::propose())
136    pub money_null_smt_snapshot: Option<SmtMemoryFp>,
137    /// Holder's instance of the Merkle tree for the `DAO` contract (holding DAO bullas)
138    pub dao_merkle_tree: MerkleTree,
139    /// Holder's instance of the Merkle tree for the `DAO` contract (holding DAO proposals)
140    pub dao_proposals_tree: MerkleTree,
141    /// Holder's set of unspent [`OwnCoin`]s from the `Money` contract
142    pub unspent_money_coins: Vec<OwnCoin>,
143    /// Holder's set of spent [`OwnCoin`]s from the `Money` contract
144    pub spent_money_coins: Vec<OwnCoin>,
145    /// Witnessed leaf positions of DAO bullas in the `dao_merkle_tree`
146    pub dao_leafs: HashMap<DaoBulla, bridgetree::Position>,
147    /// Dao Proposal snapshots
148    pub dao_prop_leafs: HashMap<DaoProposalBulla, (bridgetree::Position, MerkleTree)>,
149    /// Create bench.csv file
150    pub bench_wasm: bool,
151}

153impl Wallet {
154    /// Instantiate a new [`Wallet`] instance
155    pub async fn new(
156        keypair: Keypair,
157        token_mint_authority: Keypair,
158        contract_deploy_authority: Keypair,
159        genesis_block: BlockInfo,
160        vks: &vks::Vks,
161        verify_fees: bool,
162    ) -> Result<Self> {
163        // Create an in-memory sled db instance for this wallet
164        let sled_db = sled::Config::new().temporary(true).open()?;
165
166        // Inject the cached VKs into the database
167        vks::inject(&sled_db, vks)?;
168
169        // Create the `Validator` instance
170        let validator_config = ValidatorConfig {
171            confirmation_threshold: 3,
172            pow_target: 120,
173            pow_fixed_difficulty: Some(BigUint::from(1_u8)),
174            genesis_block,
175            verify_fees,
176        };
177        let validator = Validator::new(&sled_db, &validator_config).await?;
178
179        // The Merkle tree for the `Money` contract is initialized with a "null"
180        // leaf at position 0.
181        let mut money_merkle_tree = MerkleTree::new(1);
182        money_merkle_tree.append(MerkleNode::from(pallas::Base::ZERO));
183        money_merkle_tree.mark().unwrap();
184
185        let hasher = PoseidonFp::new();
186        let store = MemoryStorageFp::new();
187        let money_null_smt = SmtMemoryFp::new(store, hasher, &EMPTY_NODES_FP);
188
189        Ok(Self {
190            keypair,
191            token_mint_authority,
192            contract_deploy_authority,
193            validator,
194            money_merkle_tree,
195            money_null_smt,
196            money_null_smt_snapshot: None,
197            dao_merkle_tree: MerkleTree::new(1),
198            dao_proposals_tree: MerkleTree::new(1),
199            unspent_money_coins: vec![],
200            spent_money_coins: vec![],
201            dao_leafs: HashMap::new(),
202            dao_prop_leafs: HashMap::new(),
203            bench_wasm: false,
204        })
205    }
206
207    pub async fn add_transaction(
208        &mut self,
209        callname: &str,
210        tx: Transaction,
211        block_height: u32,
212    ) -> Result<()> {
213        if self.bench_wasm {
214            benchmark_wasm_calls(callname, &self.validator, &tx, block_height).await;
215        }
216
217        self.validator
218            .add_test_transactions(
219                slice::from_ref(&tx),
220                block_height,
221                self.validator.consensus.module.read().await.target,
222                true,
223                self.validator.verify_fees,
224            )
225            .await?;
226
227        // Write the data
228        {
229            let blockchain = &self.validator.blockchain;
230            let txs = &blockchain.transactions;
231            txs.insert(slice::from_ref(&tx)).expect("insert tx");
232            txs.insert_location(&[tx.hash()], block_height).expect("insert loc");
233        }
234
235        Ok(())
236    }
237}

239/// Native contract test harness instance
240pub struct TestHarness {
241    /// Initialized [`Holder`]s for this instance
242    pub holders: HashMap<Holder, Wallet>,
243    /// Cached [`ProvingKey`]s for native contract ZK proving
244    pub proving_keys: HashMap<String, (ProvingKey, ZkBinary)>,
245    /// The genesis block for this harness
246    pub genesis_block: BlockInfo,
247    /// Marker to know if we're supposed to include tx fees
248    pub verify_fees: bool,
249}
250
251impl TestHarness {
252    /// Instantiate a new [`TestHarness`] given a slice of [`Holder`]s.
253    /// Additionally, a `verify_fees` boolean will enforce tx fee verification.
254    pub async fn new(holders: &[Holder], verify_fees: bool) -> Result<Self> {
255        // Create a genesis block
256        let mut genesis_block = BlockInfo::default();
257        genesis_block.header.timestamp = Timestamp::from_u64(1689772567);
258        let producer_tx = genesis_block.txs.pop().unwrap();
259        genesis_block.append_txs(vec![producer_tx]);
260
261        // Deterministic PRNG
262        let mut rng = Pcg32::new(42);
263
264        // Build or read cached ZK PKs and VKs
265        let (pks, vks) = vks::get_cached_pks_and_vks()?;
266        let mut proving_keys = HashMap::new();
267        for (bincode, namespace, pk) in pks {
268            let mut reader = Cursor::new(pk);
269            let zkbin = ZkBinary::decode(&bincode)?;
270            let circuit = ZkCircuit::new(empty_witnesses(&zkbin)?, &zkbin);
271            let proving_key = ProvingKey::read(&mut reader, circuit)?;
272            proving_keys.insert(namespace, (proving_key, zkbin));
273        }
274
275        // Compute genesis contracts states monotree root
276        let sled_db = sled::Config::new().temporary(true).open()?;
277        vks::inject(&sled_db, &vks)?;
278        let overlay = BlockchainOverlay::new(&Blockchain::new(&sled_db)?)?;
279        deploy_native_contracts(&overlay, 90).await?;
280        genesis_block.header.state_root =
281            overlay.lock().unwrap().get_state_monotree()?.get_headroot()?.unwrap();
282
283        // Create `Wallet` instances
284        let mut holders_map = HashMap::new();
285        for holder in holders {
286            let keypair = Keypair::random(&mut rng);
287            let token_mint_authority = Keypair::random(&mut rng);
288            let contract_deploy_authority = Keypair::random(&mut rng);
289
290            let wallet = Wallet::new(
291                keypair,
292                token_mint_authority,
293                contract_deploy_authority,
294                genesis_block.clone(),
295                &vks,
296                verify_fees,
297            )
298            .await?;
299
300            holders_map.insert(*holder, wallet);
301        }
302
303        Ok(Self { holders: holders_map, proving_keys, genesis_block, verify_fees })
304    }
305
306    /// Assert that all holders' trees are the same
307    pub fn assert_trees(&self, holders: &[Holder]) {
308        assert!(holders.len() > 1);
309        // Gather wallets
310        let mut wallets = vec![];
311        for holder in holders {
312            wallets.push(self.holders.get(holder).unwrap());
313        }
314        // Compare trees
315        let wallet = wallets[0];
316        let money_root = wallet.money_merkle_tree.root(0).unwrap();
317        for wallet in &wallets[1..] {
318            assert!(money_root == wallet.money_merkle_tree.root(0).unwrap());
319        }
320    }
321}

323async fn benchmark_wasm_calls(
324    callname: &str,
325    validator: &Validator,
326    tx: &Transaction,
327    block_height: u32,
328) {
329    let mut file = std::fs::OpenOptions::new().create(true).append(true).open("bench.csv").unwrap();
330
331    for (idx, call) in tx.calls.iter().enumerate() {
332        let overlay = BlockchainOverlay::new(&validator.blockchain).expect("blockchain overlay");
333        let wasm = overlay.lock().unwrap().contracts.get(call.data.contract_id).unwrap();
334        let mut runtime = Runtime::new(
335            &wasm,
336            overlay.clone(),
337            call.data.contract_id,
338            block_height,
339            validator.consensus.module.read().await.target,
340            tx.hash(),
341            idx as u8,
342        )
343        .expect("runtime");
344
345        // Write call data
346        let mut payload = vec![];
347        tx.calls.encode(&mut payload).unwrap();
348
349        let mut times = [0; 3];
350        let now = std::time::Instant::now();
351        let _metadata = runtime.metadata(&payload).expect("metadata");
352        times[0] = now.elapsed().as_micros();
353
354        let now = std::time::Instant::now();
355        let update = runtime.exec(&payload).expect("exec");
356        times[1] = now.elapsed().as_micros();
357
358        let now = std::time::Instant::now();
359        runtime.apply(&update).expect("update");
360        times[2] = now.elapsed().as_micros();
361
362        writeln!(
363            file,
364            "{}, {}, {}, {}, {}, {}",
365            callname,
366            tx.hash(),
367            idx,
368            times[0],
369            times[1],
370            times[2]
371        )
372        .unwrap();
373    }
