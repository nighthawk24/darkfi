43    /// Create a `Money::OtcSwap` transaction with two given [`Holder`]s.
44    ///
45    /// Returns the [`Transaction`], and the transaction parameters.
46    pub async fn otc_swap(
47        &mut self,
48        holder0: &Holder,
49        owncoin0: &OwnCoin,
50        holder1: &Holder,
51        owncoin1: &OwnCoin,
52        block_height: u32,
53    ) -> Result<(Transaction, MoneyTransferParamsV1, Option<MoneyFeeParamsV1>)> {
54        let wallet0 = self.holders.get(holder0).unwrap();
55        let wallet1 = self.holders.get(holder1).unwrap();
56
57        let (mint_pk, mint_zkbin) = self.proving_keys.get(MONEY_CONTRACT_ZKAS_MINT_NS_V1).unwrap();
58        let (burn_pk, burn_zkbin) = self.proving_keys.get(MONEY_CONTRACT_ZKAS_BURN_NS_V1).unwrap();
59
60        // Use a zero spend_hook and user_data
61        let rcpt_spend_hook = FuncId::none();
62        let rcpt_user_data = pallas::Base::ZERO;
63        let rcpt_user_data_blind = Blind::random(&mut OsRng);
64
65        // Create blinding factors for commitments
66        let value_send_blind = Blind::random(&mut OsRng);
67        let value_recv_blind = Blind::random(&mut OsRng);
68        let token_send_blind = BaseBlind::random(&mut OsRng);
69        let token_recv_blind = BaseBlind::random(&mut OsRng);
70
71        // Build the first half of the swap for Holder0
72        let builder = SwapCallBuilder {
73            pubkey: wallet0.keypair.public,
74            value_send: owncoin0.note.value,
75            token_id_send: owncoin0.note.token_id,
76            value_recv: owncoin1.note.value,
77            token_id_recv: owncoin1.note.token_id,
78            user_data_blind_send: rcpt_user_data_blind,
79            spend_hook_recv: rcpt_spend_hook,
80            user_data_recv: rcpt_user_data,
81            value_blinds: [value_send_blind, value_recv_blind],
82            token_blinds: [token_send_blind, token_recv_blind],
83            coin: owncoin0.clone(),
84            tree: wallet0.money_merkle_tree.clone(),
85            mint_zkbin: mint_zkbin.clone(),
86            mint_pk: mint_pk.clone(),
87            burn_zkbin: burn_zkbin.clone(),
88            burn_pk: burn_pk.clone(),
89        };
90
91        let debris0 = builder.build()?;
92        assert!(debris0.params.inputs.len() == 1);
93        assert!(debris0.params.outputs.len() == 1);
94
95        // Build the second half of the swap for Holder1
96        let builder = SwapCallBuilder {
97            pubkey: wallet1.keypair.public,
98            value_send: owncoin1.note.value,
99            token_id_send: owncoin1.note.token_id,
100            value_recv: owncoin0.note.value,
101            token_id_recv: owncoin0.note.token_id,
102            user_data_blind_send: rcpt_user_data_blind,
103            spend_hook_recv: rcpt_spend_hook,
104            user_data_recv: rcpt_user_data,
105            value_blinds: [value_recv_blind, value_send_blind],
106            token_blinds: [token_recv_blind, token_send_blind],
107            coin: owncoin1.clone(),
108            tree: wallet1.money_merkle_tree.clone(),
109            mint_zkbin: mint_zkbin.clone(),
110            mint_pk: mint_pk.clone(),
111            burn_zkbin: burn_zkbin.clone(),
112            burn_pk: burn_pk.clone(),
113        };
114
115        let debris1 = builder.build()?;
116        assert!(debris1.params.inputs.len() == 1);
117        assert!(debris1.params.outputs.len() == 1);
118
119        // Holder1 then combines the halves
120        let swap_full_params = MoneyTransferParamsV1 {
121            inputs: vec![debris0.params.inputs[0].clone(), debris1.params.inputs[0].clone()],
122            outputs: vec![debris0.params.outputs[0].clone(), debris1.params.outputs[0].clone()],
123        };
124
125        let swap_full_proofs = vec![
126            debris0.proofs[0].clone(),
127            debris1.proofs[0].clone(),
128            debris0.proofs[1].clone(),
129            debris1.proofs[1].clone(),
130        ];
131
132        // Encode the contract call
133        let mut data = vec![MoneyFunction::OtcSwapV1 as u8];
134        swap_full_params.encode_async(&mut data).await?;
135        let call = ContractCall { contract_id: *MONEY_CONTRACT_ID, data };
136        let mut tx_builder =
137            TransactionBuilder::new(ContractCallLeaf { call, proofs: swap_full_proofs }, vec![])?;
138
139        // If we have tx fees enabled, make an offering
140        let mut fee_params = None;
141        let mut fee_signature_secrets = None;
142        if self.verify_fees {
143            let mut tx = tx_builder.build()?;
144            let sigs = tx.create_sigs(&[debris1.signature_secret])?;
145            tx.signatures = vec![sigs];
146
147            // First holder gets the partially signed transaction and adds their signature
148            let sigs = tx.create_sigs(&[debris0.signature_secret])?;
149            tx.signatures[0].insert(0, sigs[0]);
150
151            let (fee_call, fee_proofs, fee_secrets, _spent_fee_coins, fee_call_params) =
152                self.append_fee_call(holder0, tx, block_height, slice::from_ref(owncoin0)).await?;
153
154            // Append the fee call to the transaction
155            tx_builder.append(ContractCallLeaf { call: fee_call, proofs: fee_proofs }, vec![])?;
156            fee_signature_secrets = Some(fee_secrets);
157            fee_params = Some(fee_call_params);
158        }
159
160        // Now build the actual transaction and sign it with necessary keys.
161        let mut tx = tx_builder.build()?;
162        let sigs = tx.create_sigs(&[debris1.signature_secret])?;
163        tx.signatures = vec![sigs];
164        // First holder gets the partially signed transaction and adds their signature
165        let sigs = tx.create_sigs(&[debris0.signature_secret])?;
166        tx.signatures[0].insert(0, sigs[0]);
167
168        if let Some(fee_signature_secrets) = fee_signature_secrets {
169            let sigs = tx.create_sigs(&fee_signature_secrets)?;
170            tx.signatures.push(sigs);
171        }
172
173        Ok((tx, swap_full_params, fee_params))
174    }

176    /// Execute the transaction created by `otc_swap()` for a given [`Holder`].
177    ///
178    /// Returns any found [`OwnCoin`]s.
179    pub async fn execute_otc_swap_tx(
180        &mut self,
181        holder: &Holder,
182        tx: Transaction,
183        swap_params: &MoneyTransferParamsV1,
184        fee_params: &Option<MoneyFeeParamsV1>,
185        block_height: u32,
186        append: bool,
187    ) -> Result<Vec<OwnCoin>> {
188        let wallet = self.holders.get_mut(holder).unwrap();
189
190        // Execute the transaction
191        wallet.add_transaction("money::otc_swap", tx, block_height).await?;
192
193        let mut found_owncoins = vec![];
194
195        if !append {
196            return Ok(found_owncoins)
197        }
198
199        let mut inputs = swap_params.inputs.to_vec();
200        let mut outputs = swap_params.outputs.to_vec();
201        if let Some(ref fee_params) = fee_params {
202            inputs.push(fee_params.input.clone());
203            outputs.push(fee_params.output.clone());
204        }
205
206        let nullifiers = inputs.iter().map(|i| i.nullifier.inner()).map(|l| (l, l)).collect();
207        wallet.money_null_smt.insert_batch(nullifiers).expect("smt.insert_batch()");
208
209        for input in inputs {
210            if let Some(spent_coin) = wallet
211                .unspent_money_coins
212                .iter()
213                .find(|x| x.nullifier() == input.nullifier)
214                .cloned()
215            {
216                debug!("Found spent OwnCoin({}) for {:?}", spent_coin.coin, holder);
217                wallet.unspent_money_coins.retain(|x| x.nullifier() != input.nullifier);
218                wallet.spent_money_coins.push(spent_coin.clone());
219            }
220        }
221
222        for output in outputs {
223            wallet.money_merkle_tree.append(MerkleNode::from(output.coin.inner()));
224
225            // Attempt to decrypt the encrypted note
226            let Ok(note) = output.note.decrypt::<MoneyNote>(&wallet.keypair.secret) else {
227                continue
228            };
229
230            let owncoin = OwnCoin {
231                coin: output.coin,
232                note: note.clone(),
233                secret: wallet.keypair.secret,
234                leaf_position: wallet.money_merkle_tree.mark().unwrap(),
235            };
236
237            debug!("Found new OwnCoin({}) for {:?}", owncoin.coin, holder);
238            wallet.unspent_money_coins.push(owncoin.clone());
239            found_owncoins.push(owncoin);
240        }
241
242        Ok(found_owncoins)
243    }
