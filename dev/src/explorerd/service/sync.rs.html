19//! # Sync Module
20//!
21//! The `sync` module is responsible for synchronizing the explorer's database with the Darkfi
22//! blockchain network. It ensures consistency between the explorer and the blockchain by
23//! fetching missing blocks, handling reorganizations (reorgs), and subscribing to live updates
24//! through Darkfi's JSON-RPC service.
25//!
26//! ## Responsibilities
27//!
28//! - **Block Synchronization**: Handles fetching and storing blocks from a Darkfi
29//!   blockchain node during startup or when syncing, ensuring the explorer stays synchronized
30//!   with the latest confirmed blocks.
31//! - **Real-Time Updates**: Subscribes to Darkfi's JSON-RPC notification service,
32//!   allowing the explorer to process and sync new blocks as they are confirmed.
33//! - **Reorg Handling**: Detects and resolves blockchain reorganizations by identifying
34//!   the last common block (in case of divergence) and re-aligning the explorer's state with the
35//!   latest blockchain state. Reorgs are an importnt part of synchronization because they prevent
36//!   syncing invalid or outdated states, ensuring the explorer maintains an accurate view of a
37//!   Darkfi blockchain network.
38

59impl ExplorerService {
60    /// Synchronizes blocks between the explorer and a Darkfi blockchain node, ensuring
61    /// the database remains consistent by syncing any missing or outdated blocks.
62    ///
63    /// If provided `reset` is true, the explorer's blockchain-related and metric sled trees are purged
64    /// and syncing starts from the genesis block. The function also handles reorgs by re-aligning the
65    /// explorer state to the correct height when blocks are outdated. Returns a result indicating
66    /// success or failure.
67    ///
68    /// Reorg handling is delegated to the [`Self::reorg_blocks`] function, whose
69    /// documentation provides more details on the reorg process during block syncing.
70    pub async fn sync_blocks(&self, reset: bool) -> darkfi::Result<()> {
71        // Grab last synced block height from the explorer's database.
72        let last_synced_block = self.last_block().map_err(|e| {
73            let error_message = format!("[sync_blocks] Retrieving last synced block failed: {e:?}");
74            error!(target: "explorerd::rpc_blocks::sync_blocks", "{}", error_message);
75            Error::DatabaseError(error_message)
76        })?;
77
78        // Grab the last confirmed block height and hash from the darkfi node
79        let (last_darkfid_height, last_darkfid_hash) =
80            self.darkfid_client.get_last_confirmed_block().await?;
81
82        // Initialize the current height to sync from, starting from genesis block if last sync block does not exist
83        let (last_synced_height, last_synced_hash) = last_synced_block
84            .map_or((0, "".to_string()), |(height, header_hash)| (height, header_hash));
85
86        // Declare a mutable variable to track the current sync height while processing blocks
87        let mut current_height = last_synced_height;
88
89        info!(target: "explorerd::rpc_blocks::sync_blocks", "Syncing from block number: {current_height}");
90        info!(target: "explorerd::rpc_blocks::sync_blocks", "Last confirmed darkfid block: {last_darkfid_height} - {last_darkfid_hash}");
91
92        // A reorg is detected if the hash of the last synced block differs from the hash of the last confirmed block,
93        // unless the reset flag is set or the current height is 0
94        let reorg_detected = last_synced_hash != last_darkfid_hash && !reset && current_height != 0;
95
96        // If the reset flag is set, reset the explorer state and start syncing from the genesis block height.
97        // Otherwise, handle reorgs if detected, or proceed to the next block if not at the genesis height.
98        if reset {
99            self.reset_explorer_state(0)?;
100            current_height = 0;
101            info!(target: "explorerd::rpc_blocks::sync_blocks", "Reset explorer database based on set reset parameter");
102        } else if reorg_detected {
103            // Record the start time to measure the duration of potential reorg
104            let start_reorg_time = Instant::now();
105
106            // Process reorg
107            current_height = self.reorg_blocks(last_synced_height, last_darkfid_height).await?;
108
109            // Log only if a reorg occurred (i.e., the explorer wasn't merely catching up to Darkfi node blocks)
110            if current_height != last_synced_height {
111                info!(target: "explorerd::rpc_blocks::sync_blocks", "Completed reorg to height: {current_height} [{}]", fmt_duration(start_reorg_time.elapsed()));
112            }
113
114            // Prepare to sync the next block after reorg if not from genesis height
115            if current_height != 0 {
116                current_height += 1;
117            }
118        } else if current_height != 0 {
119            // Resume syncing from the block after the last synced height
120            current_height += 1;
121        }
122
123        // Record the sync start time to measure the total block sync duration
124        let sync_start_time = Instant::now();
125        // Track the number of blocks synced for reporting
126        let mut blocks_synced = 0;
127
128        // Sync blocks until the explorer is up to date with the last confirmed block
129        while current_height <= last_darkfid_height {
130            // Record the start time to measure the duration it took to sync the block
131            let block_sync_start = Instant::now();
132
133            // Retrieve the block from darkfi node by height
134            let block = match self.darkfid_client.get_block_by_height(current_height).await {
135                Ok(r) => r,
136                Err(e) => {
137                    let error_message = format!("[sync_blocks] RPC client request failed: {e:?}");
138                    error!(target: "explorerd::rpc_blocks::sync_blocks", "{}", error_message);
139                    return Err(Error::DatabaseError(error_message))
140                }
141            };
142
143            // Store the retrieved block in the explorer's database
144            if let Err(e) = self.put_block(&block).await {
145                let error_message = format!("[sync_blocks] Put block failed: {e:?}");
146                error!(target: "explorerd::rpc_blocks::sync_blocks", "{}", error_message);
147                return Err(Error::DatabaseError(error_message))
148            };
149
150            debug!(
151                target: "explorerd::rpc_blocks::sync_blocks",
152                "Synced block {current_height} [{}]",
153                fmt_duration(block_sync_start.elapsed())
154            );
155
156            // Increment the current height to sync the next block
157            current_height += 1;
158            // Increment the count of successfully synced blocks
159            blocks_synced += 1;
160        }
161
162        info!(
163            target: "explorerd::rpc_blocks::sync_blocks",
164            "Synced {blocks_synced} blocks: explorer blocks total {} [{}]",
165            self.db.blockchain.blocks.len(),
166            fmt_duration(sync_start_time.elapsed()),
167        );
168
169        Ok(())
170    }
171
172    /// Handles blockchain reorganizations (reorgs) during the explorer node's startup synchronization
173    /// with Darkfi nodes, ensuring the explorer provides a consistent and accurate view of the blockchain.
174    ///
175    /// A reorg occurs when the blocks stored by the blockchain nodes diverge from those stored by the explorer.
176    /// This function resolves inconsistencies by identifying the point of divergence, searching backward through
177    /// block heights, and comparing block hashes between the explorer database and the blockchain node. Once a
178    /// common block height is found, the explorer is re-aligned to that height.
179    ///
180    /// If no common block can be found, the explorer resets to the "genesis height," removing all blocks,
181    /// transactions, and metrics from its database to resynchronize with the canonical chain from the nodes.
182    ///
183    /// Returns the last height at which the explorer's state was successfully re-aligned with the blockchain.
184    async fn reorg_blocks(
185        &self,
186        last_synced_height: u32,
187        last_darkfid_height: u32,
188    ) -> darkfi::Result<u32> {
189        // Log reorg detection in the case that explorer height is greater or equal to height of darkfi node
190        if last_synced_height >= last_darkfid_height {
191            info!(target: "explorerd::rpc_blocks::process_sync_blocks_reorg",
192                "Reorg detected with heights: explorer.{last_synced_height} >= darkfid.{last_darkfid_height}");
193        }
194
195        // Declare a mutable variable to track the current height while searching for a common block
196        let mut cur_height = last_synced_height;
197        // Search for an explorer block that matches a darkfi node block
198        while cur_height > 0 {
199            let synced_block = self.get_block_by_height(cur_height)?;
200            debug!(target: "explorerd::rpc_blocks::process_sync_blocks_reorg", "Searching for common block: {cur_height}");
201
202            // Check if we found a synced block for current height being searched
203            if let Some(synced_block) = synced_block {
204                // Fetch the block from darkfi node to check for a match
205                match self.darkfid_client.get_block_by_height(cur_height).await {
206                    Ok(darkfid_block) => {
207                        // If hashes match, we've found the point of divergence
208                        if synced_block.header_hash == darkfid_block.hash().to_string() {
209                            // If hashes match but the cur_height differs from the last synced height, reset the explorer state
210                            if cur_height != last_synced_height {
211                                self.reset_explorer_state(cur_height)?;
212                                debug!(target: "explorerd::rpc_blocks::process_sync_blocks_reorg", "Completed reorg to height: {cur_height}");
213                            }
214                            break;
215                        } else {
216                            // Log reorg detection with height and header hash mismatch details
217                            if cur_height == last_synced_height {
218                                info!(
219                                    target: "explorerd::rpc_blocks::process_sync_blocks_reorg",
220                                    "Reorg detected at height {cur_height}: explorer.{} != darkfid.{}",
221                                    synced_block.header_hash,
222                                    darkfid_block.hash()
223                                );
224                            }
225                        }
226                    }
227                    // Continue searching for blocks that do not exist on darkfi nodes
228                    Err(Error::JsonRpcError((-32121, _))) => (),
229                    Err(e) => {
230                        let error_message =
231                            format!("[process_sync_blocks_reorg] RPC client request failed: {e:?}");
232                        error!(target: "explorerd::rpc_blocks::process_sync_blocks_reorg", "{}", error_message);
233                        return Err(Error::DatabaseError(error_message))
234                    }
235                }
236            }
237
238            // Move to previous block to search for a match
239            cur_height = cur_height.saturating_sub(1);
240        }
241
242        // Check if genesis block reorg is needed
243        if cur_height == 0 {
244            self.reset_explorer_state(0)?;
245        }
246
247        // Return the last height we reorged to
248        Ok(cur_height)
249    }
250}
251/// Subscribes to darkfid's JSON-RPC notification endpoint that serves
252/// new confirmed blocks. Upon receiving them, store them to the database.
253pub async fn subscribe_sync_blocks(
254    explorer: Arc<Explorerd>,
255    endpoint: Url,
256    ex: Arc<smol::Executor<'static>>,
257) -> darkfi::Result<(StoppableTaskPtr, StoppableTaskPtr)> {
258    // Grab last confirmed block
259    let (last_darkfid_height, last_darkfid_hash) =
260        explorer.darkfid_client.get_last_confirmed_block().await?;
261
262    // Grab last synced block
263    let (mut height, hash) = match explorer.service.last_block() {
264        Ok(Some((height, hash))) => (height, hash),
265        Ok(None) => (0, "".to_string()),
266        Err(e) => {
267            return Err(Error::DatabaseError(format!(
268                "[subscribe_blocks] Retrieving last synced block failed: {e:?}"
269            )))
270        }
271    };
272
273    // Evaluates whether there is a mismatch between the last confirmed block and the last synced block
274    let blocks_mismatch = (last_darkfid_height != height || last_darkfid_hash != hash) &&
275        last_darkfid_height != 0 &&
276        height != 0;
277
278    // Check if there is a mismatch, throwing an error to prevent operating in a potentially inconsistent state
279    if blocks_mismatch {
280        warn!(target: "explorerd::rpc_blocks::subscribe_blocks",
281        "Warning: Last synced block is not the last confirmed block: \
282        last_darkfid_height={last_darkfid_height}, last_synced_height={height}, last_darkfid_hash={last_darkfid_hash}, last_synced_hash={hash}");
283        warn!(target: "explorerd::rpc_blocks::subscribe_blocks", "You should first fully sync the blockchain, and then subscribe");
284        return Err(Error::DatabaseError(
285            "[subscribe_blocks] Blockchain not fully synced".to_string(),
286        ));
287    }
288
289    info!(target: "explorerd::rpc_blocks::subscribe_blocks", "Subscribing to receive notifications of incoming blocks");
290    let publisher = Publisher::new();
291    let subscription = publisher.clone().subscribe().await;
292    let _ex = ex.clone();
293    let subscriber_task = StoppableTask::new();
294    subscriber_task.clone().start(
295        // Weird hack to prevent lifetimes hell
296        async move {
297            let ex = _ex.clone();
298            let rpc_client = RpcClient::new(endpoint, ex).await?;
299            let req = JsonRequest::new("blockchain.subscribe_blocks", JsonValue::Array(vec![]));
300            rpc_client.subscribe(req, publisher).await
301        },
302        |res| async move {
303            match res {
304                Ok(()) => { /* Do nothing */ }
305                Err(e) => error!(target: "explorerd::rpc_blocks::subscribe_blocks", "[subscribe_blocks] JSON-RPC server error: {e:?}"),
306            }
307        },
308        Error::RpcServerStopped,
309        ex.clone(),
310    );
311    info!(target: "explorerd::rpc_blocks::subscribe_blocks", "Detached subscription to background");
312
313    let listener_task = StoppableTask::new();
314    listener_task.clone().start(
315        // Weird hack to prevent lifetimes hell
316        async move {
317            loop {
318                match subscription.receive().await {
319                    JsonResult::Notification(n) => {
320                        debug!(target: "explorerd::rpc_blocks::subscribe_blocks", "Got Block notification from darkfid subscription");
321                        if n.method != "blockchain.subscribe_blocks" {
322                            return Err(Error::UnexpectedJsonRpc(format!(
323                                "Got foreign notification from darkfid: {}",
324                                n.method
325                            )))
326                        }
327
328                        // Verify parameters
329                        if !n.params.is_array() {
330                            return Err(Error::UnexpectedJsonRpc(
331                                "Received notification params are not an array".to_string(),
332                            ))
333                        }
334                        let params = n.params.get::<Vec<JsonValue>>().unwrap();
335                        if params.is_empty() {
336                            return Err(Error::UnexpectedJsonRpc(
337                                "Notification parameters are empty".to_string(),
338                            ))
339                        }
340
341                        for param in params {
342                            let param = param.get::<String>().unwrap();
343                            let bytes = base64::decode(param).unwrap();
344
345                            let darkfid_block: BlockInfo = match deserialize_async(&bytes).await {
346                                Ok(b) => b,
347                                Err(e) => {
348                                    return Err(Error::UnexpectedJsonRpc(format!(
349                                        "[subscribe_blocks] Deserializing block failed: {e:?}"
350                                    )))
351                                },
352                            };
353                            info!(target: "explorerd::rpc_blocks::subscribe_blocks", "========================================================================================");
354                            info!(target: "explorerd::rpc_blocks::subscribe_blocks", "| Block Notification: {} |", darkfid_block.hash());
355                            info!(target: "explorerd::rpc_blocks::subscribe_blocks", "========================================================================================");
356
357                            // Store darkfi node block height for later use
358                            let darkfid_block_height = darkfid_block.header.height;
359
360                            // Check if we need to perform a reorg due to mismatch in block heights
361                            if darkfid_block_height <= height {
362                                info!(target: "explorerd::rpc_blocks::subscribe_blocks",
363                                    "Reorg detected with heights: darkfid.{darkfid_block_height} <= explorer.{height}");
364
365                                // Calculate the reset height
366                                let reset_height = darkfid_block_height.saturating_sub(1);
367
368                                // Record the start time to measure the duration of the reorg
369                                let start_reorg_time = Instant::now();
370
371                                // Execute the reorg by resetting the explorer state to reset height
372                                explorer.service.reset_explorer_state(reset_height)?;
373                                info!(target: "explorerd::rpc_blocks::subscribe_blocks", "Completed reorg to height: {reset_height} [{}]", fmt_duration(start_reorg_time.elapsed()));
374                            }
375
376
377                            // Record the start time to measure the duration to store the block
378                            let start_reorg_time = Instant::now();
379
380                            if let Err(e) = explorer.service.put_block(&darkfid_block).await {
381                                return Err(Error::DatabaseError(format!(
382                                    "[subscribe_blocks] Put block failed: {e:?}"
383                                )))
384                            }
385
386                            info!(target: "explorerd::rpc_blocks::subscribe_blocks", "Stored new block at height: {} [{}]", darkfid_block.header.height, fmt_duration(start_reorg_time.elapsed()));
387
388                            // Process the next block
389                            height = darkfid_block.header.height;
390                        }
391                    }
392
393                    JsonResult::Error(e) => {
394                        // Some error happened in the transmission
395                        return Err(Error::UnexpectedJsonRpc(format!("Got error from JSON-RPC: {e:?}")))
396                    }
397
398                    x => {
399                        // And this is weird
400                        return Err(Error::UnexpectedJsonRpc(format!(
401                            "Got unexpected data from JSON-RPC: {x:?}"
402                        )))
403                    }
404                }
405            };
406        },
407        |res| async move {
408            match res {
409                Ok(()) => { /* Do nothing */ }
410                Err(e) => error!(target: "explorerd::rpc_blocks::subscribe_blocks", "[subscribe_blocks] JSON-RPC server error: {e:?}"),
411            }
412        },
413        Error::RpcServerStopped,
414        ex,
415    );
416
417    Ok((subscriber_task, listener_task))
