19use std::{io::Result, str::Chars};
20
21use darkfi_serial::{serialize, VarInt};
22
23use super::{
24    ast::{Arg, Constant, Literal, Statement, StatementType, Witness},
25    error::ErrorEmitter,
26    types::HeapType,
27};
28
29/// Version of the binary
30pub const BINARY_VERSION: u8 = 2;
31/// Magic bytes prepended to the binary
32pub const MAGIC_BYTES: [u8; 4] = [0x0b, 0x01, 0xb1, 0x35];
33
34pub struct Compiler {
35    namespace: String,
36    k: u32,
37    constants: Vec<Constant>,
38    witnesses: Vec<Witness>,
39    statements: Vec<Statement>,
40    literals: Vec<Literal>,
41    debug_info: bool,
42    error: ErrorEmitter,
43}
44
45impl Compiler {
46    #[allow(clippy::too_many_arguments)]
47    pub fn new(
48        filename: &str,
49        source: Chars,
50        namespace: String,
51        k: u32,
52        constants: Vec<Constant>,
53        witnesses: Vec<Witness>,
54        statements: Vec<Statement>,
55        literals: Vec<Literal>,
56        debug_info: bool,
57    ) -> Self {
58        // For nice error reporting, we'll load everything into a string
59        // vector so we have references to lines.
60        let lines: Vec<String> = source.as_str().lines().map(|x| x.to_string()).collect();
61        let error = ErrorEmitter::new("Compiler", filename, lines);
62
63        Self { namespace, k, constants, witnesses, statements, literals, debug_info, error }
64    }
65
66    pub fn compile(&self) -> Result<Vec<u8>> {
67        let mut bincode = vec![];
68
69        // Write the magic bytes and version
70        bincode.extend_from_slice(&MAGIC_BYTES);
71        bincode.push(BINARY_VERSION);
72
73        // Write the circuit's k param
74        bincode.extend_from_slice(&serialize(&self.k));
75
76        // Write the circuit's namespace
77        bincode.extend_from_slice(&serialize(&self.namespace));
78
79        // Temporary heap vector for lookups
80        let mut tmp_heap = vec![];
81
82        // In the .constant section of the binary, we write the constant's type,
83        // and the name so the VM can look it up from `src/crypto/constants/`.
84        bincode.extend_from_slice(b".constant");
85        for i in &self.constants {
86            tmp_heap.push(i.name.as_str());
87            bincode.push(i.typ as u8);
88            bincode.extend_from_slice(&serialize(&i.name));
89        }
90
91        // Currently, our literals are only Uint64 types, in the binary we'll
92        // add them here in the .literal section. In the VM, they will be on
93        // their own heap, used for reference by opcodes.
94        bincode.extend_from_slice(b".literal");
95        for i in &self.literals {
96            bincode.push(i.typ as u8);
97            bincode.extend_from_slice(&serialize(&i.name));
98        }
99
100        // In the .witness section, we write all our witness types, on the heap
101        // they're in order of appearance.
102        bincode.extend_from_slice(b".witness");
103        for i in &self.witnesses {
104            tmp_heap.push(i.name.as_str());
105            bincode.push(i.typ as u8);
106        }
107
108        bincode.extend_from_slice(b".circuit");
109        for i in &self.statements {
110            match i.typ {
111                StatementType::Assign => tmp_heap.push(&i.lhs.as_ref().unwrap().name),
112                // In case of a simple call, we don't append anything to the heap
113                StatementType::Call => {}
114                _ => unreachable!("Invalid statement type in circuit: {:?}", i.typ),
115            }
116
117            bincode.push(i.opcode as u8);
118            bincode.extend_from_slice(&serialize(&VarInt(i.rhs.len() as u64)));
119
120            for arg in &i.rhs {
121                match arg {
122                    Arg::Var(arg) => {
123                        if let Some(found) = Compiler::lookup_heap(&tmp_heap, &arg.name) {
124                            bincode.push(HeapType::Var as u8);
125                            bincode.extend_from_slice(&serialize(&VarInt(found as u64)));
126                            continue
127                        }
128
129                        return Err(self.error.abort(
130                            &format!("Failed finding a heap reference for `{}`", arg.name),
131                            arg.line,
132                            arg.column,
133                        ))
134                    }
135                    Arg::Lit(lit) => {
136                        if let Some(found) = Compiler::lookup_literal(&self.literals, &lit.name) {
137                            bincode.push(HeapType::Lit as u8);
138                            bincode.extend_from_slice(&serialize(&VarInt(found as u64)));
139                            continue
140                        }
141
142                        return Err(self.error.abort(
143                            &format!("Failed finding literal `{}`", lit.name),
144                            lit.line,
145                            lit.column,
146                        ))
147                    }
148                    _ => unreachable!(),
149                };
150            }
151        }
152
153        // If we're not doing debug info, we're done here and can return.
154        if !self.debug_info {
155            return Ok(bincode)
156        }
157
158        // TODO: Otherwise, we proceed appending debug info.
159
160        Ok(bincode)
161    }
162
163    fn lookup_heap(heap: &[&str], name: &str) -> Option<usize> {
164        for (idx, n) in heap.iter().enumerate() {
165            if n == &name {
166                return Some(idx)
167            }
168        }
169
170        None
171    }
172
173    fn lookup_literal(literals: &[Literal], name: &str) -> Option<usize> {
174        for (idx, n) in literals.iter().enumerate() {
175            if n.name == name {
176                return Some(idx)
177            }
178        }
179
180        None
181    }
182}
