81/// An Event Graph instance
82pub struct EventGraph {
83    /// Pointer to the P2P network instance
84    p2p: P2pPtr,
85    /// Sled tree containing the DAG
86    dag: sled::Tree,
87    /// Replay logs path.
88    datastore: PathBuf,
89    /// Run in replay_mode where if set we log Sled DB instructions
90    /// into `datastore`, useful to reacreate a faulty DAG to debug.
91    replay_mode: bool,
92    /// The set of unreferenced DAG tips
93    unreferenced_tips: RwLock<BTreeMap<u64, HashSet<blake3::Hash>>>,
94    /// A `HashSet` containg event IDs and their 1-level parents.
95    /// These come from the events we've sent out using `EventPut`.
96    /// They are used with `EventReq` to decide if we should reply
97    /// or not. Additionally it is also used when we broadcast the
98    /// `TipRep` message telling peers about our unreferenced tips.
99    broadcasted_ids: RwLock<HashSet<blake3::Hash>>,
100    /// DAG Pruning Task
101    pub prune_task: OnceCell<StoppableTaskPtr>,
102    /// Event publisher, this notifies whenever an event is
103    /// inserted into the DAG
104    pub event_pub: PublisherPtr<Event>,
105    /// Current genesis event
106    current_genesis: RwLock<Event>,
107    /// Currently configured DAG rotation, in days
108    days_rotation: u64,
109    /// Flag signalling DAG has finished initial sync
110    pub synced: RwLock<bool>,
111    /// Enable graph debugging
112    pub deg_enabled: RwLock<bool>,
113    /// The publisher for which we can give deg info over
114    deg_publisher: PublisherPtr<DegEvent>,
115}

117impl EventGraph {
118    /// Create a new [`EventGraph`] instance, creates a new Genesis
119    /// event and checks if it
120    /// is containd in DAG, if not prunes DAG, may also start a pruning
121    /// task based on `days_rotation`, and return an atomic instance of
122    /// `Self`
123    /// * `p2p` atomic pointer to p2p.
124    /// * `sled_db` sled DB instance.
125    /// * `datastore` path where we should log db instrucion if run in
126    ///   replay mode.
127    /// * `replay_mode` set the flag to keep a log of db instructions.
128    /// * `dag_tree_name` the name of disk-backed tree (or DAG name).
129    /// * `days_rotation` marks the lifetime of the DAG before it's
130    ///   pruned.
131    pub async fn new(
132        p2p: P2pPtr,
133        sled_db: sled::Db,
134        datastore: PathBuf,
135        replay_mode: bool,
136        dag_tree_name: &str,
137        days_rotation: u64,
138        ex: Arc<Executor<'_>>,
139    ) -> Result<EventGraphPtr> {
140        let dag = sled_db.open_tree(dag_tree_name)?;
141        let unreferenced_tips = RwLock::new(BTreeMap::new());
142        let broadcasted_ids = RwLock::new(HashSet::new());
143        let event_pub = Publisher::new();
144
145        // Create the current genesis event based on the `days_rotation`
146        let current_genesis = generate_genesis(days_rotation);
147        let self_ = Arc::new(Self {
148            p2p,
149            dag: dag.clone(),
150            datastore,
151            replay_mode,
152            unreferenced_tips,
153            broadcasted_ids,
154            prune_task: OnceCell::new(),
155            event_pub,
156            current_genesis: RwLock::new(current_genesis.clone()),
157            days_rotation,
158            synced: RwLock::new(false),
159            deg_enabled: RwLock::new(false),
160            deg_publisher: Publisher::new(),
161        });
162
163        // Check if we have it in our DAG.
164        // If not, we can prune the DAG and insert this new genesis event.
165        if !dag.contains_key(current_genesis.id().as_bytes())? {
166            info!(
167                target: "event_graph::new()",
168                "[EVENTGRAPH] DAG does not contain current genesis, pruning existing data",
169            );
170            self_.dag_prune(current_genesis).await?;
171        }
172
173        // Find the unreferenced tips in the current DAG state.
174        *self_.unreferenced_tips.write().await = self_.find_unreferenced_tips().await;
175
176        // Spawn the DAG pruning task
177        if days_rotation > 0 {
178            let prune_task = StoppableTask::new();
179            let _ = self_.prune_task.set(prune_task.clone()).await;
180
181            prune_task.clone().start(
182                self_.clone().dag_prune_task(days_rotation),
183                |res| async move {
184                    match res {
185                        Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
186                        Err(e) => error!(target: "event_graph::_handle_stop()", "[EVENTGRAPH] Failed stopping prune task: {e}")
187                    }
188                },
189                Error::DetachedTaskStopped,
190                ex.clone(),
191            );
192        }
193
194        Ok(self_)
195    }

198        self.days_rotation

201    /// Sync the DAG from connected peers
202    pub async fn dag_sync(&self) -> Result<()> {
203        // We do an optimistic sync where we ask all our connected peers for
204        // the latest layer DAG tips (unreferenced events) and then we accept
205        // the ones we see the most times.
206        // * Compare received tips with local ones, identify which we are missing.
207        // * Request these from peers
208        // * Recursively request these backward
209        //
210        // Verification:
211        // * Timestamps should go backwards
212        // * Cross-check with multiple peers, this means we should request the
213        //   same event from multiple peers and make sure it is the same.
214        // * Since we should be pruning, if we're not synced after some reasonable
215        //   amount of iterations, these could be faulty peers and we can try again
216        //   from the beginning
217
218        // Get references to all our peers.
219        let channels = self.p2p.hosts().peers();
220        let mut communicated_peers = channels.len();
221        info!(
222            target: "event_graph::dag_sync()",
223            "[EVENTGRAPH] Syncing DAG from {communicated_peers} peers..."
224        );
225
226        // Here we keep track of the tips, their layers and how many time we've seen them.
227        let mut tips: HashMap<blake3::Hash, (u64, usize)> = HashMap::new();
228
229        // Let's first ask all of our peers for their tips and collect them
230        // in our hashmap above.
231        for channel in channels.iter() {
232            let url = channel.display_address();
233
234            let tip_rep_sub = match channel.subscribe_msg::<TipRep>().await {
235                Ok(v) => v,
236                Err(e) => {
237                    error!(
238                        target: "event_graph::dag_sync()",
239                        "[EVENTGRAPH] Sync: Couldn't subscribe TipReq for peer {url}, skipping ({e})"
240                    );
241                    communicated_peers -= 1;
242                    continue
243                }
244            };
245
246            if let Err(e) = channel.send(&TipReq {}).await {
247                error!(
248                    target: "event_graph::dag_sync()",
249                    "[EVENTGRAPH] Sync: Couldn't contact peer {url}, skipping ({e})"
250                );
251                communicated_peers -= 1;
252                continue
253            };
254
255            // Node waits for response
256            let Ok(peer_tips) = tip_rep_sub
257                .receive_with_timeout(self.p2p.settings().read().await.outbound_connect_timeout)
258                .await
259            else {
260                error!(
261                    target: "event_graph::dag_sync()",
262                    "[EVENTGRAPH] Sync: Peer {url} didn't reply with tips in time, skipping"
263                );
264                communicated_peers -= 1;
265                continue
266            };
267
268            let peer_tips = &peer_tips.0;
269
270            // Note down the seen tips
271            for (layer, layer_tips) in peer_tips {
272                for tip in layer_tips {
273                    if let Some(seen_tip) = tips.get_mut(tip) {
274                        seen_tip.1 += 1;
275                    } else {
276                        tips.insert(*tip, (*layer, 1));
277                    }
278                }
279            }
280        }
281
282        // After we've communicated all the peers, let's see what happened.
283        if tips.is_empty() {
284            error!(
285                target: "event_graph::dag_sync()",
286                "[EVENTGRAPH] Sync: Could not find any DAG tips",
287            );
288            return Err(Error::DagSyncFailed)
289        }
290
291        // We know the number of peers we've communicated with,
292        // so we will consider events we saw at more than 2/3 of
293        // those peers.
294        let consideration_threshold = communicated_peers * 2 / 3;
295        let mut considered_tips = HashSet::new();
296        for (tip, (_, amount)) in tips.iter() {
297            if amount > &consideration_threshold {
298                considered_tips.insert(*tip);
299            }
300        }
301        drop(tips);
302
303        // Now begin fetching the events backwards.
304        let mut missing_parents = HashSet::new();
305        for tip in considered_tips.iter() {
306            assert!(tip != &NULL_ID);
307
308            if !self.dag.contains_key(tip.as_bytes()).unwrap() {
309                missing_parents.insert(*tip);
310            }
311        }
312
313        if missing_parents.is_empty() {
314            *self.synced.write().await = true;
315            info!(target: "event_graph::dag_sync()", "[EVENTGRAPH] DAG synced successfully!");
316            return Ok(())
317        }
318
319        info!(target: "event_graph::dag_sync()", "[EVENTGRAPH] Fetching events");
320        let mut received_events: BTreeMap<u64, Vec<Event>> = BTreeMap::new();
321        let mut received_events_hashes = HashSet::new();
322
323        while !missing_parents.is_empty() {
324            let mut found_event = false;
325
326            for channel in channels.iter() {
327                let url = channel.display_address();
328
329                debug!(
330                    target: "event_graph::dag_sync()",
331                    "Requesting {missing_parents:?} from {url}..."
332                );
333
334                let ev_rep_sub = match channel.subscribe_msg::<EventRep>().await {
335                    Ok(v) => v,
336                    Err(e) => {
337                        error!(
338                            target: "event_graph::dag_sync()",
339                            "[EVENTGRAPH] Sync: Couldn't subscribe EventRep for peer {url}, skipping ({e})"
340                        );
341                        continue
342                    }
343                };
344
345                let request_missing_events = missing_parents.clone().into_iter().collect();
346                if let Err(e) = channel.send(&EventReq(request_missing_events)).await {
347                    error!(
348                        target: "event_graph::dag_sync()",
349                        "[EVENTGRAPH] Sync: Failed communicating EventReq({missing_parents:?}) to {url}: {e}"
350                    );
351                    continue
352                }
353
354                // Node waits for response
355                let Ok(parent) = ev_rep_sub
356                    .receive_with_timeout(self.p2p.settings().read().await.outbound_connect_timeout)
357                    .await
358                else {
359                    error!(
360                        target: "event_graph::dag_sync()",
361                        "[EVENTGRAPH] Sync: Timeout waiting for parents {missing_parents:?} from {url}"
362                    );
363                    continue
364                };
365
366                let parents = parent.0.clone();
367
368                for parent in parents {
369                    let parent_id = parent.id();
370                    if !missing_parents.contains(&parent_id) {
371                        error!(
372                            target: "event_graph::dag_sync()",
373                            "[EVENTGRAPH] Sync: Peer {url} replied with a wrong event: {}",
374                            parent.id()
375                        );
376                        continue
377                    }
378
379                    debug!(
380                        target: "event_graph::dag_sync()",
381                        "Got correct parent event {parent_id}"
382                    );
383
384                    if let Some(layer_events) = received_events.get_mut(&parent.layer) {
385                        layer_events.push(parent.clone());
386                    } else {
387                        let layer_events = vec![parent.clone()];
388                        received_events.insert(parent.layer, layer_events);
389                    }
390                    received_events_hashes.insert(parent_id);
391
392                    missing_parents.remove(&parent_id);
393                    found_event = true;
394
395                    // See if we have the upper parents
396                    for upper_parent in parent.parents.iter() {
397                        if upper_parent == &NULL_ID {
398                            continue
399                        }
400
401                        if !missing_parents.contains(upper_parent) &&
402                            !received_events_hashes.contains(upper_parent) &&
403                            !self.dag.contains_key(upper_parent.as_bytes()).unwrap()
404                        {
405                            debug!(
406                                target: "event_graph::dag_sync()",
407                                "Found upper missing parent event {upper_parent}"
408                            );
409                            missing_parents.insert(*upper_parent);
410                        }
411                    }
412                }
413
414                break
415            }
416
417            if !found_event {
418                error!(
419                    target: "event_graph::dag_sync()",
420                    "[EVENTGRAPH] Sync: Failed to get all events",
421                );
422                return Err(Error::DagSyncFailed)
423            }
424        } // <-- while !missing_parents.is_empty
425
426        // At this point we should've got all the events.
427        // We should add them to the DAG.
428        let mut events = vec![];
429        for (_, tips) in received_events {
430            for tip in tips {
431                events.push(tip);
432            }
433        }
434        self.dag_insert(&events).await?;
435
436        *self.synced.write().await = true;
437
438        info!(target: "event_graph::dag_sync()", "[EVENTGRAPH] DAG synced successfully!");
439        Ok(())
440    }

442    /// Atomically prune the DAG and insert the given event as genesis.
443    async fn dag_prune(&self, genesis_event: Event) -> Result<()> {
444        debug!(target: "event_graph::dag_prune()", "Pruning DAG...");
445
446        // Acquire exclusive locks to unreferenced_tips, broadcasted_ids and
447        // current_genesis while this operation is happening. We do this to
448        // ensure that during the pruning operation, no other operations are
449        // able to access the intermediate state which could lead to producing
450        // the wrong state after pruning.
451        let mut unreferenced_tips = self.unreferenced_tips.write().await;
452        let mut broadcasted_ids = self.broadcasted_ids.write().await;
453        let mut current_genesis = self.current_genesis.write().await;
454
455        // Atomically clear the DAG and write the new genesis event.
456        let mut batch = sled::Batch::default();
457        for key in self.dag.iter().keys() {
458            batch.remove(key.unwrap());
459        }
460        batch.insert(genesis_event.id().as_bytes(), serialize_async(&genesis_event).await);
461
462        debug!(target: "event_graph::dag_prune()", "Applying batch...");
463        if let Err(e) = self.dag.apply_batch(batch) {
464            panic!("Failed pruning DAG, sled apply_batch error: {e}");
465        }
466
467        // Clear unreferenced tips and bcast ids
468        *unreferenced_tips = BTreeMap::new();
469        unreferenced_tips.insert(0, HashSet::from([genesis_event.id()]));
470        *current_genesis = genesis_event;
471        *broadcasted_ids = HashSet::new();
472        drop(unreferenced_tips);
473        drop(broadcasted_ids);
474        drop(current_genesis);
475
476        debug!(target: "event_graph::dag_prune()", "DAG pruned successfully");
477        Ok(())
478    }

480    /// Background task periodically pruning the DAG.
481    async fn dag_prune_task(self: Arc<Self>, days_rotation: u64) -> Result<()> {
482        // The DAG should periodically be pruned. This can be a configurable
483        // parameter. By pruning, we should deterministically replace the
484        // genesis event (can use a deterministic timestamp) and drop everything
485        // in the DAG, leaving just the new genesis event.
486        debug!(target: "event_graph::dag_prune_task()", "Spawned background DAG pruning task");
487
488        loop {
489            // Find the next rotation timestamp:
490            let next_rotation = next_rotation_timestamp(INITIAL_GENESIS, days_rotation);
491
492            // Prepare the new genesis event
493            let current_genesis = Event {
494                timestamp: next_rotation,
495                content: GENESIS_CONTENTS.to_vec(),
496                parents: [NULL_ID; N_EVENT_PARENTS],
497                layer: 0,
498            };
499
500            // Sleep until it's time to rotate.
501            let s = millis_until_next_rotation(next_rotation);
502
503            debug!(target: "event_graph::dag_prune_task()", "Sleeping {s}ms until next DAG prune");
504            msleep(s).await;
505            debug!(target: "event_graph::dag_prune_task()", "Rotation period reached");
506
507            // Trigger DAG prune
508            self.dag_prune(current_genesis).await?;
509        }
510    }

512    /// Atomically insert given events into the DAG and return the event IDs.
513    /// All provided events must be valid. An overlay is used over the DAG tree,
514    /// temporary writting each event in order. After all events have been
515    /// validated and inserted successfully, we write the overlay to sled.
516    /// This will append the new events into the unreferenced tips set, and
517    /// remove the events' parents from it. It will also append the events'
518    /// level-1 parents to the `broadcasted_ids` set, so the P2P protocol
519    /// knows that any requests for them are actually legitimate.
520    /// TODO: The `broadcasted_ids` set should periodically be pruned, when
521    /// some sensible time has passed after broadcasting the event.
522    pub async fn dag_insert(&self, events: &[Event]) -> Result<Vec<blake3::Hash>> {
523        // Sanity check
524        if events.is_empty() {
525            return Ok(vec![])
526        }
527
528        // Acquire exclusive locks to `unreferenced_tips and broadcasted_ids`
529        let mut unreferenced_tips = self.unreferenced_tips.write().await;
530        let mut broadcasted_ids = self.broadcasted_ids.write().await;
531
532        // Here we keep the IDs to return
533        let mut ids = Vec::with_capacity(events.len());
534
535        // Create an overlay over the DAG tree
536        let mut overlay = SledTreeOverlay::new(&self.dag);
537
538        // Grab genesis timestamp
539        let genesis_timestamp = self.current_genesis.read().await.timestamp;
540
541        // Iterate over given events to validate them and
542        // write them to the overlay
543        for event in events {
544            let event_id = event.id();
545            debug!(
546                target: "event_graph::dag_insert()",
547                "Inserting event {event_id} into the DAG"
548            );
549
550            if !event
551                .validate(&self.dag, genesis_timestamp, self.days_rotation, Some(&overlay))
552                .await?
553            {
554                error!(target: "event_graph::dag_insert()", "Event {event_id} is invalid!");
555                return Err(Error::EventIsInvalid)
556            }
557
558            let event_se = serialize_async(event).await;
559
560            // Add the event to the overlay
561            overlay.insert(event_id.as_bytes(), &event_se)?;
562
563            if self.replay_mode {
564                replayer_log(&self.datastore, "insert".to_owned(), event_se)?;
565            }
566            // Note down the event ID to return
567            ids.push(event_id);
568        }
569
570        // Aggregate changes into a single batch
571        let batch = overlay.aggregate().unwrap();
572
573        // Atomically apply the batch.
574        // Panic if something is corrupted.
575        if let Err(e) = self.dag.apply_batch(batch) {
576            panic!("Failed applying dag_insert batch to sled: {e}");
577        }
578
579        // Iterate over given events to update references and
580        // send out notifications about them
581        for event in events {
582            let event_id = event.id();
583
584            // Update the unreferenced DAG tips set
585            debug!(
586                target: "event_graph::dag_insert()",
587                "Event {event_id} parents {:#?}", event.parents,
588            );
589            for parent_id in event.parents.iter() {
590                if parent_id != &NULL_ID {
591                    debug!(
592                        target: "event_graph::dag_insert()",
593                        "Removing {parent_id} from unreferenced_tips"
594                    );
595
596                    // Iterate over unreferenced tips in previous layers
597                    // and remove the parent
598                    // NOTE: this might be too exhaustive, but the
599                    // assumption is that previous layers unreferenced
600                    // tips will be few.
601                    for (layer, tips) in unreferenced_tips.iter_mut() {
602                        if layer >= &event.layer {
603                            continue
604                        }
605                        tips.remove(parent_id);
606                    }
607                    broadcasted_ids.insert(*parent_id);
608                }
609            }
610            unreferenced_tips.retain(|_, tips| !tips.is_empty());
611            debug!(
612                target: "event_graph::dag_insert()",
613                "Adding {event_id} to unreferenced tips"
614            );
615
616            if let Some(layer_tips) = unreferenced_tips.get_mut(&event.layer) {
617                layer_tips.insert(event_id);
618            } else {
619                let mut layer_tips = HashSet::new();
620                layer_tips.insert(event_id);
621                unreferenced_tips.insert(event.layer, layer_tips);
622            }
623
624            // Send out notifications about the new event
625            self.event_pub.notify(event.clone()).await;
626        }
627
628        // Drop the exclusive locks
629        drop(unreferenced_tips);
630        drop(broadcasted_ids);
631
632        Ok(ids)
633    }

635    /// Fetch an event from the DAG

643    /// Get next layer along with its N_EVENT_PARENTS from the unreferenced
644    /// tips of the DAG. Since tips are mapped by their layer, we go backwards
645    /// until we fill the vector, ensuring we always use latest layers tips as
646    /// parents.
647    async fn get_next_layer_with_parents(&self) -> (u64, [blake3::Hash; N_EVENT_PARENTS]) {
648        let unreferenced_tips = self.unreferenced_tips.read().await;
649
650        let mut parents = [NULL_ID; N_EVENT_PARENTS];
651        let mut index = 0;
652        'outer: for (_, tips) in unreferenced_tips.iter().rev() {
653            for tip in tips.iter() {
654                parents[index] = *tip;
655                index += 1;
656                if index >= N_EVENT_PARENTS {
657                    break 'outer
658                }
659            }
660        }
661
662        let next_layer = unreferenced_tips.last_key_value().unwrap().0 + 1;
663
664        assert!(parents.iter().any(|x| x != &NULL_ID));
665        (next_layer, parents)
666    }
667
668    /// Find the unreferenced tips in the current DAG state, mapped by their layers.
669    async fn find_unreferenced_tips(&self) -> BTreeMap<u64, HashSet<blake3::Hash>> {
670        // First get all the event IDs
671        let mut tips = HashSet::new();
672        for iter_elem in self.dag.iter() {
673            let (id, _) = iter_elem.unwrap();
674            let id = blake3::Hash::from_bytes((&id as &[u8]).try_into().unwrap());
675            tips.insert(id);
676        }
677
678        // Iterate again to find unreferenced IDs
679        for iter_elem in self.dag.iter() {
680            let (_, event) = iter_elem.unwrap();
681            let event: Event = deserialize_async(&event).await.unwrap();
682            for parent in event.parents.iter() {
683                tips.remove(parent);
684            }
685        }
686
687        // Build the layers map
688        let mut map: BTreeMap<u64, HashSet<blake3::Hash>> = BTreeMap::new();
689        for tip in tips {
690            let event = self.dag_get(&tip).await.unwrap().unwrap();
691            if let Some(layer_tips) = map.get_mut(&event.layer) {
692                layer_tips.insert(tip);
693            } else {
694                let mut layer_tips = HashSet::new();
695                layer_tips.insert(tip);
696                map.insert(event.layer, layer_tips);
697            }
698        }
699
700        map
701    }
702
703    /// Internal function used for DAG sorting.
704    async fn get_unreferenced_tips_sorted(&self) -> [blake3::Hash; N_EVENT_PARENTS] {
705        let (_, tips) = self.get_next_layer_with_parents().await;
706
707        // Convert the hash to BigUint for sorting
708        let mut sorted: Vec<_> =
709            tips.iter().map(|x| BigUint::from_bytes_be(x.as_bytes())).collect();
710        sorted.sort_unstable();
711
712        // Convert back to blake3
713        let mut tips_sorted = [NULL_ID; N_EVENT_PARENTS];
714        for (i, id) in sorted.iter().enumerate() {
715            let mut bytes = id.to_bytes_be();
716
717            // Ensure we have 32 bytes
718            while bytes.len() < blake3::OUT_LEN {
719                bytes.insert(0, 0);
720            }
721
722            tips_sorted[i] = blake3::Hash::from_bytes(bytes.try_into().unwrap());
723        }
724
725        tips_sorted
726    }

728    /// Perform a topological sort of the DAG.
729    pub async fn order_events(&self) -> Vec<Event> {
730        let mut ordered_events = VecDeque::new();
731        let mut visited = HashSet::new();
732
733        for tip in self.get_unreferenced_tips_sorted().await {
734            if !visited.contains(&tip) && tip != NULL_ID {
735                let tip = self.dag_get(&tip).await.unwrap().unwrap();
736                ordered_events.extend(self.dfs_topological_sort(tip, &mut visited).await);
737            }
738        }
739
740        let mut ord_events_vec = ordered_events.make_contiguous().to_vec();
741        // Order events based on thier layer numbers, or based on timestamp if they are equal
742        ord_events_vec
743            .sort_unstable_by(|a, b| a.0.cmp(&b.0).then(b.1.timestamp.cmp(&a.1.timestamp)));
744
745        ord_events_vec.iter().map(|a| a.1.clone()).collect::<Vec<Event>>()
746    }

748    /// We do a non-recursive DFS (<https://en.wikipedia.org/wiki/Depth-first_search>),
749    /// and additionally we consider the timestamps.
750    async fn dfs_topological_sort(
751        &self,
752        event: Event,
753        visited: &mut HashSet<blake3::Hash>,
754    ) -> VecDeque<(u64, Event)> {
755        let mut ordered_events = VecDeque::new();
756        let mut stack = VecDeque::new();
757        let event_id = event.id();
758        stack.push_back(event_id);
759
760        while let Some(event_id) = stack.pop_front() {
761            if !visited.contains(&event_id) && event_id != NULL_ID {
762                visited.insert(event_id);
763                if let Some(event) = self.dag_get(&event_id).await.unwrap() {
764                    for parent in event.parents.iter() {
765                        stack.push_back(*parent);
766                    }
767
768                    ordered_events.push_back((event.layer, event))
769                }
770            }
771        }
772
773        ordered_events
774    }

798    pub async fn eventgraph_info(&self, id: u16, _params: JsonValue) -> JsonResult {
799        let mut graph = HashMap::new();
800        for iter_elem in self.dag.iter() {
801            let (id, val) = iter_elem.unwrap();
802            let id = blake3::Hash::from_bytes((&id as &[u8]).try_into().unwrap());
803            let val: Event = deserialize_async(&val).await.unwrap();
804            graph.insert(id, val);
805        }
806
807        let json_graph = graph
808            .into_iter()
809            .map(|(k, v)| {
810                let key = k.to_string();
811                let value = JsonValue::from(v);
812                (key, value)
813            })
814            .collect();
815        let values = json_map([("dag", JsonValue::Object(json_graph))]);
816
817        let result = JsonValue::Object(HashMap::from([("eventgraph_info".to_string(), values)]));
818
819        JsonResponse::new(result, id).into()
820    }

822    /// Fetch all the events that are on a higher layers than the
823    /// provided ones.
824    pub async fn fetch_successors_of(
825        &self,
826        tips: BTreeMap<u64, HashSet<blake3::Hash>>,
827    ) -> Result<Vec<Event>> {
828        debug!(
829             target: "event_graph::fetch_successors_of()",
830             "fetching successors of {tips:?}"
831        );
832
833        let mut graph = HashMap::new();
834        for iter_elem in self.dag.iter() {
835            let (id, val) = iter_elem.unwrap();
836            let hash = blake3::Hash::from_bytes((&id as &[u8]).try_into().unwrap());
837            let event: Event = deserialize_async(&val).await.unwrap();
838            graph.insert(hash, event);
839        }
840
841        let mut result = vec![];
842
843        'outer: for tip in tips.iter() {
844            for i in tip.1.iter() {
845                if !graph.contains_key(i) {
846                    continue 'outer;
847                }
848            }
849
850            for (_, ev) in graph.iter() {
851                if ev.layer > *tip.0 && !result.contains(ev) {
852                    result.push(ev.clone())
853                }
854            }
855        }
856
857        result.sort_by(|a, b| a.layer.cmp(&b.layer));
858
859        Ok(result)
860    }
