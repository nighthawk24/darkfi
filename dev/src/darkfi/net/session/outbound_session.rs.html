19//! Outbound connections session. Manages the creation of outbound sessions.
20//! Used to create an outbound session and to stop and start the session.
21//!
22//! Class consists of a weak pointer to the p2p interface and a vector of
23//! outbound connection slots. Using a weak pointer to p2p allows us to
24//! avoid circular dependencies. The vector of slots is wrapped in a mutex
25//! lock. This is switched on every time we instantiate a connection slot
26//! and insures that no other part of the program uses the slots at the
27//! same time.

62/// Defines outbound connections session.
63pub struct OutboundSession {
64    /// Weak pointer to parent p2p object
65    pub(in crate::net) p2p: Weak<P2p>,
66    /// Outbound connection slots
67    slots: Mutex<Vec<Arc<Slot>>>,
68    /// Peer discovery task
69    peer_discovery: Arc<PeerDiscovery>,
70}
71
72impl OutboundSession {
73    /// Create a new outbound session.
74    pub(crate) fn new(p2p: Weak<P2p>) -> OutboundSessionPtr {
75        Arc::new_cyclic(|session| Self {
76            p2p,
77            slots: Mutex::new(Vec::new()),
78            peer_discovery: PeerDiscovery::new(session.clone()),
79        })
80    }
81
82    /// Start the outbound session. Runs the channel connect loop.
83    pub(crate) async fn start(self: Arc<Self>) {
84        let n_slots = self.p2p().settings().read().await.outbound_connections;
85        verbose!(target: "net::outbound_session", "[P2P] Starting {n_slots} outbound connection slots.");
86
87        // Activate mutex lock on connection slots.
88        let mut slots = self.slots.lock().await;
89
90        let mut futures = FuturesUnordered::new();
91
92        let self_ = Arc::downgrade(&self);
93
94        for i in 0..n_slots as u32 {
95            let slot = Slot::new(self_.clone(), i);
96            futures.push(slot.clone().start());
97            slots.push(slot);
98        }
99
100        while (futures.next().await).is_some() {}
101
102        self.peer_discovery.clone().start().await;
103    }
104
105    /// Stops the outbound session.
106    pub(crate) async fn stop(&self) {
107        debug!(target: "net::outbound_session", "Stopping outbound session..");
108        let slots = &*self.slots.lock().await;
109        let mut futures = FuturesUnordered::new();
110
111        for slot in slots {
112            futures.push(slot.clone().stop());
113        }
114
115        while (futures.next().await).is_some() {}
116
117        self.peer_discovery.clone().stop().await;
118        debug!(target: "net::outbound_session", "Outbound session stopped!");
119    }
120
121    pub async fn slot_info(&self) -> Vec<u32> {
122        let mut info = Vec::new();
123        let slots = &*self.slots.lock().await;
124        for slot in slots {
125            info.push(slot.channel_id.load(Ordering::Relaxed));
126        }
127        info
128    }
129
130    fn wakeup_peer_discovery(&self) {
131        self.peer_discovery.notify()
132    }
133
134    async fn wakeup_slots(&self) {
135        let slots = &*self.slots.lock().await;
136        for slot in slots {
137            slot.notify();
138        }
139    }
140}

153struct Slot {
154    slot: u32,
155    process: StoppableTaskPtr,
156    wakeup_self: CondVar,
157    session: Weak<OutboundSession>,
158    connector: Connector,
159    // For debugging
160    channel_id: AtomicU32,
161}
162
163impl Slot {
164    fn new(session: Weak<OutboundSession>, slot: u32) -> Arc<Self> {
165        let settings = session.upgrade().unwrap().p2p().settings();
166
167        Arc::new(Self {
168            slot,
169            process: StoppableTask::new(),
170            wakeup_self: CondVar::new(),
171            session: session.clone(),
172            connector: Connector::new(settings, session),
173            channel_id: AtomicU32::new(0),
174        })
175    }
176
177    async fn start(self: Arc<Self>) {
178        let ex = self.p2p().executor();
179
180        self.process.clone().start(
181            self.run(),
182            |res| async {
183                match res {
184                    Ok(()) | Err(Error::NetworkServiceStopped) => {}
185                    Err(e) => error!("net::outbound_session {e}"),
186                }
187            },
188            Error::NetworkServiceStopped,
189            ex,
190        );
191    }
192
193    async fn stop(self: Arc<Self>) {
194        self.connector.stop();
195        self.process.stop().await;
196    }
197
198    /// Address selection algorithm that works as follows: up to
199    /// gold_count, select from the goldlist. Up to white_count,
200    /// select from the whitelist. For all other slots, select from
201    /// the greylist. If none of these preferences are satisfied, do
202    /// peer discovery.
203    ///
204    /// Selecting from the greylist for some % of the slots is necessary
205    /// and healthy since we require the network retains some unreliable
206    /// connections. A network that purely favors uptime over unreliable
207    /// connections may be vulnerable to sybil by attackers with good uptime.
208    async fn fetch_addrs(&self) -> Option<(Url, u64)> {
209        let hosts = self.p2p().hosts();
210        let slot = self.slot as usize;
211        let container = &self.p2p().hosts().container;
212
213        // Acquire Settings read lock
214        let settings = self.p2p().settings().read_arc().await;
215
216        let white_count = (settings.white_connect_percent * settings.outbound_connections) / 100;
217        let gold_count = settings.gold_connect_count;
218
219        let transports = settings.allowed_transports.clone();
220        let preference_strict = settings.slot_preference_strict;
221
222        // Drop Settings read lock
223        drop(settings);
224
225        let grey_only = hosts.container.is_empty(HostColor::White) &&
226            hosts.container.is_empty(HostColor::Gold) &&
227            !hosts.container.is_empty(HostColor::Grey);
228
229        // If we only have grey entries, select from the greylist. Otherwise,
230        // use the preference defined in settings.
231        let addrs = if grey_only && !preference_strict {
232            container.fetch_with_schemes(HostColor::Grey as usize, &transports, None)
233        } else if slot < gold_count {
234            container.fetch_with_schemes(HostColor::Gold as usize, &transports, None)
235        } else if slot < white_count {
236            container.fetch_with_schemes(HostColor::White as usize, &transports, None)
237        } else {
238            container.fetch_with_schemes(HostColor::Grey as usize, &transports, None)
239        };
240
241        hosts.check_addrs(addrs).await
242    }
243
244    // We first try to make connections to the addresses on our gold list. We then find some
245    // whitelist connections according to the whitelist percent default. Finally, any remaining
246    // connections we make from the greylist.
247    async fn run(self: Arc<Self>) -> Result<()> {
248        let hosts = self.p2p().hosts();
249
250        loop {
251            // Activate the slot
252            debug!(
253                target: "net::outbound_session::try_connect()",
254                "[P2P] Finding a host to connect to for outbound slot #{}",
255                self.slot,
256            );
257
258            // Do peer discovery if we don't have any peers on the Grey, White or Gold list
259            // (first time connecting to the network).
260            if hosts.container.is_empty(HostColor::Grey) &&
261                hosts.container.is_empty(HostColor::White) &&
262                hosts.container.is_empty(HostColor::Gold)
263            {
264                dnetev!(self, OutboundSlotSleeping, {
265                    slot: self.slot,
266                });
267
268                self.wakeup_self.reset();
269                // Peer discovery
270                self.session().wakeup_peer_discovery();
271                // Wait to be woken up by peer discovery
272                self.wakeup_self.wait().await;
273
274                continue
275            }
276
277            let addr = if let Some(addr) = self.fetch_addrs().await {
278                debug!(target: "net::outbound_session::run()", "Fetched addr={}, slot #{}", addr.0,
279                self.slot);
280                addr
281            } else {
282                debug!(target: "net::outbound_session::run()", "No address found! Activating peer discovery...");
283                dnetev!(self, OutboundSlotSleeping, {
284                    slot: self.slot,
285                });
286
287                self.wakeup_self.reset();
288                // Peer discovery
289                self.session().wakeup_peer_discovery();
290                // Wait to be woken up by peer discovery
291                self.wakeup_self.wait().await;
292
293                continue
294            };
295
296            let host = addr.0;
297            let last_seen = addr.1;
298            let slot = self.slot;
299
300            verbose!(
301                target: "net::outbound_session::try_connect()",
302                "[P2P] Connecting outbound slot #{slot} [{host}]"
303            );
304
305            dnetev!(self, OutboundSlotConnecting, {
306                slot,
307                addr: host.clone(),
308            });
309
310            let (_, channel) = match self.try_connect(host.clone(), last_seen).await {
311                Ok(connect_info) => connect_info,
312                Err(err) => {
313                    debug!(
314                        target: "net::outbound_session::try_connect()",
315                        "[P2P] Outbound slot #{slot} connection failed: {err}"
316                    );
317
318                    dnetev!(self, OutboundSlotDisconnected, {
319                        slot,
320                        err: err.to_string()
321                    });
322
323                    self.channel_id.store(0, Ordering::Relaxed);
324
325                    continue
326                }
327            };
328
329            // At this point we've managed to connect.
330            let stop_sub = channel.subscribe_stop().await?;
331
332            verbose!(
333                target: "net::outbound_session::try_connect()",
334                "[P2P] Outbound slot #{slot} connected [{}]",
335                channel.display_address()
336            );
337
338            dnetev!(self, OutboundSlotConnected, {
339                slot: self.slot,
340                addr: channel.display_address().clone(),
341                channel_id: channel.info.id
342            });
343
344            // Setup new channel
345            if let Err(err) =
346                self.session().register_channel(channel.clone(), self.p2p().executor()).await
347            {
348                verbose!(
349                    target: "net::outbound_session",
350                    "[P2P] Outbound slot #{slot} disconnected: {err}"
351                );
352
353                dnetev!(self, OutboundSlotDisconnected, {
354                    slot: self.slot,
355                    err: err.to_string()
356                });
357
358                self.channel_id.store(0, Ordering::Relaxed);
359
360                warn!(
361                    target: "net::outbound_session::try_connect()",
362                    "[P2P] Suspending addr=[{}] slot #{slot}",
363                    channel.display_address()
364                );
365
366                // Peer disconnected during the registry process. We'll downgrade this peer now.
367                if let Err(e) = self
368                    .p2p()
369                    .hosts()
370                    .move_host(channel.address(), last_seen, HostColor::Grey)
371                    .await
372                {
373                    warn!(target: "net::outbound_session", "Error while moving addr={} to greylist: {e}", channel.display_address());
374                    continue
375                }
376
377                // Mark its state as Suspend, which sends this node to the Refinery for processing.
378                if let Err(e) =
379                    self.p2p().hosts().try_register(channel.address().clone(), HostState::Suspend)
380                {
381                    warn!(target: "net::outbound_session", "Error while suspending addr={}: {e}", channel.display_address());
382                }
383
384                continue
385            }
386
387            self.channel_id.store(channel.info.id, Ordering::Relaxed);
388
389            // Wait for channel to close
390            stop_sub.receive().await;
391
392            self.channel_id.store(0, Ordering::Relaxed);
393        }
394    }
395
396    /// Start making an outbound connection, using provided [`Connector`].
397    /// Tries to find a valid address to connect to, otherwise does peer
398    /// discovery. The peer discovery loops until some peer we can connect
399    /// to is found. Once connected, registers the channel, removes it from
400    /// the list of pending channels, and starts sending messages across the
401    /// channel. In case of any failures, a network error is returned and the
402    /// main connect loop (parent of this function) will iterate again.
403    async fn try_connect(&self, addr: Url, last_seen: u64) -> Result<(Url, ChannelPtr)> {
404        match self.connector.connect(&addr).await {
405            Ok((addr_final, channel)) => Ok((addr_final, channel)),
406
407            Err(err) => {
408                verbose!(
409                    target: "net::outbound_session::try_connect()",
410                    "[P2P] Unable to connect outbound slot #{} {err}",
411                    self.slot
412                );
413
414                // Immediately return if the Connector has stopped.
415                // This indicates a shutdown of the P2P network and
416                // should not result in hostlist modifications.
417                if let Error::ConnectorStopped(message) = err {
418                    return Err(Error::ConnectFailed(message));
419                }
420
421                // At this point we failed to connect. We'll downgrade this peer now.
422                self.p2p().hosts().move_host(&addr, last_seen, HostColor::Grey).await?;
423
424                // Mark its state as Suspend, which sends it to the Refinery for processing.
425                if let Err(e) = self.p2p().hosts().try_register(addr.clone(), HostState::Suspend) {
426                    warn!(target: "net::outbound_session::try_connect()", "Error while suspending addr={addr}: {e}");
427                }
428
429                // Notify that channel processing failed
430                self.p2p().hosts().channel_publisher.notify(Err(err.clone())).await;
431
432                Err(err)
433            }
434        }
435    }
436
437    fn notify(&self) {
438        self.wakeup_self.notify()
439    }

441    fn session(&self) -> OutboundSessionPtr {
442        self.session.upgrade().unwrap()
443    }
444    fn p2p(&self) -> P2pPtr {
445        self.session().p2p()
446    }

462
463    async fn wait(&self) -> bool;

465    fn notify(&self);
466

468

470}

472/// Main PeerDiscovery process that loops through connected peers
473/// and sends out a `GetAddrs` when it is active. If there are no
474/// connected peers after two attempts, connect to our seed nodes
475/// and perform `SeedSyncSession`.
476struct PeerDiscovery {
477    process: StoppableTaskPtr,
478    wakeup_self: CondVar,
479    session: Weak<OutboundSession>,
480}
481
482impl PeerDiscovery {
483    fn new(session: Weak<OutboundSession>) -> Arc<Self> {
484        Arc::new(Self { process: StoppableTask::new(), wakeup_self: CondVar::new(), session })
485    }
486}
487
488#[async_trait]
489impl PeerDiscoveryBase for PeerDiscovery {
490    async fn start(self: Arc<Self>) {
491        let ex = self.p2p().executor();
492        self.process.clone().start(
493            async move {
494                self.run().await;
495                unreachable!();
496            },
497            // Ignore stop handler
498            |_| async {},
499            Error::NetworkServiceStopped,
500            ex,
501        );
502    }
503    async fn stop(self: Arc<Self>) {
504        self.process.stop().await;
505    }
506
507    /// Activate peer discovery if not active already. For the first two
508    /// attempts, this will loop through all connected P2P peers and send
509    /// out a `GetAddrs` message to request more peers. Other parts of the
510    /// P2P stack will then handle the incoming addresses and place them in
511    /// the hosts list.  
512    ///
513    /// On the third attempt, and if we still haven't made any connections,
514    /// this function will then call `p2p.seed()` which triggers a
515    /// `SeedSyncSession` that will connect to configured seeds and request
516    /// peers from them.
517    ///
518    /// This function will also sleep `outbound_peer_discovery_attempt_time`
519    /// seconds after broadcasting in order to let the P2P stack receive and
520    /// work through the addresses it is expecting.
521    async fn run(self: Arc<Self>) {
522        let mut current_attempt = 0;
523        loop {
524            dnetev!(self, OutboundPeerDiscovery, {
525                attempt: current_attempt,
526                state: "wait",
527            });
528
529            // wait to be woken up by notify()
530            let sleep_was_instant = self.wait().await;
531
532            // Read the current P2P settings
533            let settings = self.p2p().settings().read_arc().await;
534            let outbound_peer_discovery_cooloff_time =
535                settings.outbound_peer_discovery_cooloff_time;
536            let outbound_peer_discovery_attempt_time =
537                settings.outbound_peer_discovery_attempt_time;
538            let outbound_connections = settings.outbound_connections;
539            let getaddrs_max = settings.getaddrs_max;
540            let allowed_transports = settings.allowed_transports.clone();
541            let seeds = settings.seeds.clone();
542            drop(settings);
543
544            if sleep_was_instant {
545                // Try again
546                current_attempt += 1;
547            } else {
548                // reset back to start
549                current_attempt = 1;
550            }
551
552            if current_attempt >= 4 {
553                verbose!(
554                    target: "net::outbound_session::peer_discovery()",
555                    "[P2P] [PEER DISCOVERY] Sleeping and trying again. Attempt {current_attempt}"
556                );
557
558                dnetev!(self, OutboundPeerDiscovery, {
559                    attempt: current_attempt,
560                    state: "sleep",
561                });
562
563                sleep(outbound_peer_discovery_cooloff_time).await;
564                current_attempt = 1;
565            }
566
567            // First 2 times try sending GetAddr to the network.
568            // 3rd time do a seed sync (providing we have seeds
569            // configured).
570            if self.p2p().is_connected() && current_attempt <= 2 {
571                // Broadcast the GetAddrs message to all active peers.
572                // If we have no active peers, we will perform a SeedSyncSession instead.
573                verbose!(
574                    target: "net::outbound_session::peer_discovery()",
575                    "[P2P] [PEER DISCOVERY] Asking peers for new peers to connect to...");
576
577                dnetev!(self, OutboundPeerDiscovery, {
578                    attempt: current_attempt,
579                    state: "getaddr",
580                });
581
582                let get_addrs = GetAddrsMessage {
583                    max: getaddrs_max.unwrap_or(outbound_connections.min(u32::MAX as usize) as u32),
584                    transports: allowed_transports,
585                };
586
587                self.p2p().broadcast(&get_addrs).await;
588
589                // Wait for a hosts store update event
590                let store_sub = self.p2p().hosts().subscribe_store().await;
591
592                let result = timeout(
593                    Duration::from_secs(outbound_peer_discovery_attempt_time),
594                    store_sub.receive(),
595                )
596                .await;
597
598                match result {
599                    Ok(addrs_len) => {
600                        verbose!(
601                            target: "net::outbound_session::peer_discovery()",
602                            "[P2P] [PEER DISCOVERY] Discovered {addrs_len} peers"
603                        );
604                    }
605                    Err(_) => {
606                        warn!(
607                            target: "net::outbound_session::peer_discovery()",
608                            "[P2P] [PEER DISCOVERY] Waiting for addrs timed out."
609                        );
610                        // Just do seed next time
611                        current_attempt = 3;
612                    }
613                }
614
615                // NOTE: not every call to subscribe() in net/ has a
616                // corresponding unsubscribe(). To do this we need async
617                // Drop. For now it's sufficient for publishers to be
618                // de-allocated when the Session completes.
619                store_sub.unsubscribe().await;
620            } else if !seeds.is_empty() {
621                verbose!(
622                    target: "net::outbound_session::peer_discovery()",
623                    "[P2P] [PEER DISCOVERY] Asking seeds for new peers to connect to...");
624
625                dnetev!(self, OutboundPeerDiscovery, {
626                    attempt: current_attempt,
627                    state: "seed",
628                });
629
630                self.p2p().seed().await;
631            }
632
633            self.wakeup_self.reset();
634            self.session().wakeup_slots().await;
635
636            // Give some time for new connections to be established
637            sleep(outbound_peer_discovery_attempt_time).await;
638        }
639    }
640
641    /// Blocks execution until we receive a notification from notify().
642    /// `wakeup_self.wait()` resets the condition variable (`CondVar`) and waits
643    /// for a call from `notify()`. Returns `true` if the function completed
644    /// instantly (i.e. no wait occured). Returns false otherwise.
645    async fn wait(&self) -> bool {
646        let wakeup_start = Instant::now();
647        self.wakeup_self.wait().await;
648        let wakeup_end = Instant::now();
649
650        let epsilon = Duration::from_millis(200);
651        wakeup_end - wakeup_start <= epsilon
652    }
653
654    /// Wakeup peer discovery by sending a notification to `wakeup_self`.
655    /// Uses the underlying `CondVar` method `notify()`. Subsequent calls
656    /// to this do nothing until `wait()` is called.
657    fn notify(&self) {
658        self.wakeup_self.notify()
659    }
660
661    fn session(&self) -> OutboundSessionPtr {
662        self.session.upgrade().unwrap()
663    }
664
665    fn p2p(&self) -> P2pPtr {
666        self.session().p2p()
667    }
