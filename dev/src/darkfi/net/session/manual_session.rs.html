19//! Manual connections session. Manages the creation of manual sessions.
20//! Used to create a manual session and to stop and start the session.
21//!
22//! A manual session is a type of outbound session in which we attempt
23//! connection to a predefined set of peers. Manual sessions loop forever
24//! continually trying to connect to a given peer, and sleep
25//! `outbound_connect_timeout` times between each attempt.
26//!
27//! Class consists of a weak pointer to the p2p interface and a vector of
28//! outbound connection slots. Using a weak pointer to p2p allows us to
29//! avoid circular dependencies. The vector of slots is wrapped in a mutex
30//! lock. This is switched on every time we instantiate a connection slot
31//! and insures that no other part of the program uses the slots at the
32//! same time.
33
34use std::sync::{Arc, Weak};

36use async_trait::async_trait;
37use futures::stream::{FuturesUnordered, StreamExt};
38use smol::lock::{Mutex as AsyncMutex, RwLock as AsyncRwLock};
39use tracing::{debug, error, warn};
40use url::Url;
41
42use super::{
43    super::{
44        connector::Connector,
45        p2p::{P2p, P2pPtr},
46    },
47    Session, SessionBitFlag, SESSION_MANUAL,
48};
49use crate::{
50    net::{hosts::HostState, settings::Settings},
51    system::{sleep, StoppableTask, StoppableTaskPtr},
52    util::logger::verbose,
53    Error, Result,
54};

56pub type ManualSessionPtr = Arc<ManualSession>;
57
58/// Defines manual connections session.
59pub struct ManualSession {
60    pub(in crate::net) p2p: Weak<P2p>,
61    slots: AsyncMutex<Vec<Arc<Slot>>>,
62}
63
64impl ManualSession {
65    /// Create a new manual session.
66    pub fn new(p2p: Weak<P2p>) -> ManualSessionPtr {
67        Arc::new(Self { p2p, slots: AsyncMutex::new(Vec::new()) })
68    }
69
70    pub(crate) async fn start(self: Arc<Self>) {
71        // Activate mutex lock on connection slots.
72        let mut slots = self.slots.lock().await;
73
74        let mut futures = FuturesUnordered::new();
75
76        let self_ = Arc::downgrade(&self);
77
78        // Initialize a slot for each configured peer.
79        // Connections will be started by not yet activated.
80        for peer in &self.p2p().settings().read().await.peers {
81            let slot = Slot::new(self_.clone(), peer.clone(), self.p2p().settings());
82            futures.push(slot.clone().start());
83            slots.push(slot);
84        }
85
86        while (futures.next().await).is_some() {}
87    }
88
89    /// Stops the manual session.
90    pub async fn stop(&self) {
91        let slots = &*self.slots.lock().await;
92        let mut futures = FuturesUnordered::new();
93
94        for slot in slots {
95            futures.push(slot.stop());
96        }
97
98        while (futures.next().await).is_some() {}
99    }
100}
101
102#[async_trait]
103impl Session for ManualSession {
104    fn p2p(&self) -> P2pPtr {
105        self.p2p.upgrade().unwrap()
106    }
107
108    fn type_id(&self) -> SessionBitFlag {
109        SESSION_MANUAL
110    }
111}
112
113struct Slot {
114    addr: Url,
115    process: StoppableTaskPtr,
116    session: Weak<ManualSession>,
117    connector: Connector,
118}
119
120impl Slot {
121    fn new(
122        session: Weak<ManualSession>,
123        addr: Url,
124        settings: Arc<AsyncRwLock<Settings>>,
125    ) -> Arc<Self> {
126        Arc::new(Self {
127            addr,
128            process: StoppableTask::new(),
129            session: session.clone(),
130            connector: Connector::new(settings, session),
131        })
132    }
133
134    async fn start(self: Arc<Self>) {
135        let ex = self.p2p().executor();
136
137        self.process.clone().start(
138            self.run(),
139            |res| async {
140                match res {
141                    Ok(()) | Err(Error::NetworkServiceStopped) => {}
142                    Err(e) => error!("net::manual_session {e}"),
143                }
144            },
145            Error::NetworkServiceStopped,
146            ex,
147        );
148    }
149
150    /// Attempts a connection on the associated Connector object.
151    async fn run(self: Arc<Self>) -> Result<()> {
152        let ex = self.p2p().executor();
153
154        let mut attempts = 0;
155        loop {
156            attempts += 1;
157
158            verbose!(
159                target: "net::manual_session",
160                "[P2P] Connecting to manual outbound [{}] (attempt #{})",
161                self.addr, attempts
162            );
163
164            let settings = self.p2p().settings().read_arc().await;
165            let seeds = settings.seeds.clone();
166            let outbound_connect_timeout = settings.outbound_connect_timeout;
167            drop(settings);
168
169            // Do not establish a connection to a host that is also configured as a seed.
170            // This indicates a user misconfiguration.
171            if seeds.contains(&self.addr) {
172                error!(
173                    target: "net::manual_session",
174                    "[P2P] Suspending manual connection to seed [{}]", self.addr.clone(),
175                );
176                return Ok(())
177            }
178
179            if let Err(e) = self.p2p().hosts().try_register(self.addr.clone(), HostState::Connect) {
180                debug!(target: "net::manual_session",
181                    "Cannot connect to manual={}, err={e}", &self.addr);
182
183                sleep(outbound_connect_timeout).await;
184
185                continue
186            }
187
188            match self.connector.connect(&self.addr).await {
189                Ok((_, channel)) => {
190                    verbose!(
191                        target: "net::manual_session",
192                        "[P2P] Manual outbound connected [{}]",
193                        channel.display_address()
194                    );
195
196                    let stop_sub = channel.subscribe_stop().await?;
197
198                    // Channel is now connected but not yet setup
199
200                    // Register the new channel
201                    match self.session().register_channel(channel.clone(), ex.clone()).await {
202                        Ok(()) => {
203                            // Wait for channel to close
204                            stop_sub.receive().await;
205
206                            verbose!(
207                                target: "net::manual_session",
208                                "[P2P] Manual outbound disconnected [{}]",
209                                channel.display_address()
210                            );
211                        }
212                        Err(e) => {
213                            warn!(
214                                target: "net::manual_session",
215                                "[P2P] Unable to connect to manual outbound [{}]: {e}",
216                                channel.display_address(),
217                            );
218
219                            // Free up this addr for future operations.
220                            if let Err(e) = self.p2p().hosts().unregister(channel.address()) {
221                                warn!(target: "net::manual_session", "[P2P] Error while unregistering addr={}, err={e}", channel.display_address());
222                            }
223                        }
224                    }
225                }
226                Err(e) => {
227                    warn!(
228                        target: "net::manual_session",
229                        "[P2P] Unable to connect to manual outbound: {e}",
230                    );
231
232                    // Free up this addr for future operations.
233                    if let Err(e) = self.p2p().hosts().unregister(&self.addr) {
234                        warn!(target: "net::manual_session", "[P2P] Error while unregistering addr={}, err={e}", self.addr);
235                    }
236                }
237            }
238
239            verbose!(
240                target: "net::manual_session",
241                "[P2P] Waiting {outbound_connect_timeout} seconds until next manual outbound connection attempt [{}]",
242                self.addr,
243            );
244
245            sleep(outbound_connect_timeout).await;
246        }
247    }
248
249    fn session(&self) -> ManualSessionPtr {
250        self.session.upgrade().unwrap()
251    }
252
253    fn p2p(&self) -> P2pPtr {
254        self.session().p2p()
255    }
256
257    async fn stop(&self) {
258        self.connector.stop();
259        self.process.stop().await;
260    }
261}
